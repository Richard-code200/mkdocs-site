基本操作命令

创建仓库

# 在当前目录初始化新仓库
git init

# 克隆远程仓库
git clone <repository-url>
git clone <repository-url> <directory-name>  # 克隆到指定目录

查看状态

# 查看工作区状态
git status
git status -s  # 简短格式输出

# 查看文件差异
git diff              # 工作区 vs 暂存区
git diff --staged     # 暂存区 vs 最新提交
git diff HEAD         # 工作区 vs 最新提交
git diff <commit1> <commit2>  # 比较两个提交

添加文件到暂存区

# 添加单个文件
git add <filename>

# 添加多个文件
git add file1.txt file2.txt

# 添加所有文件(包括新文件和修改)
git add .
git add -A
git add --all

# 添加所有已跟踪文件的修改
git add -u

# 交互式添加(选择性暂存)
git add -p

提交更改

# 提交暂存区内容
git commit -m "提交说明"

# 跳过暂存区,直接提交所有已跟踪文件的修改
git commit -a -m "提交说明"

# 修改最后一次提交(修改提交信息或补文件)
git commit --amend
git commit --amend -m "新的提交说明"

查看历史

# 查看提交历史
git log
git log --oneline             # 一行显示
git log --graph               # 图形化显示分支
git log --all                 # 显示所有分支
git log --author="username"   # 按作者筛选
git log --since="2026-3-20"   # 按时间筛选
git log --grep="关键词"       # 搜索提交信息
git log -p                    # 显示详细修改内容
git log --stat                # 显示文件修改统计
git log -5                    # 显示最近5条提交历史

# 查看某个文件的修改历史
git log <filename>
git log -p <filename>         # 显示具体修改

# 查看谁修改了文件的每一行
git blame <filename>
